FILL NULL WITH GLOBAL MEAN / AVG

Calculate the average from non-null values and use it to replace missing numeric values.

WITH salary_average AS (
    SELECT AVG(salary) AS avg_salary

FILL NULL WITH GROUP AVERAGE

Fill each missing salary using the average salary from the employee's department.

SQL / PostgreSQL 
SELECT
    employee_id,
    department,
    COALESCE(
        salary,
        AVG(salary) OVER (
            PARTITION BY department
        )
    ) AS salary
FROM employees;

UDF & SQL Registration

PYSPARK UDF: UPPERCASE NAME + SQL REGISTRATION

Built-in functions such as F.upper are faster. This example shows the exact Python UDF and SQL registration pattern requested.

SQL / PostgreSQL 
-- Prefer the built-in SQL function
SELECT
    customer_id,
    UPPER(customer_name)
        AS formatted_name
FROM customers;

-- Optional PostgreSQL function
CREATE OR REPLACE FUNCTION format_name(
    input_name TEXT
)
RETURNS TEXT
LANGUAGE SQL
IMMUTABLE
AS $$
    SELECT UPPER(input_name)
$$;
PySpark 
from pyspark.sql.functions import udf
from pyspark.sql.types import (
    StringType,
    IntegerType
)

def format_name(customer_name):
    if customer_name is None:
        return None
    return customer_name.upper()

# Convert Python function to PySpark UDF
format_name_udf = udf(
    format_name,
    StringType()
)

result = customers_df.withColumn(
    "formatted_name",
    format_name_udf(
        F.col("customer_name")
    )
)

# Register the UDF for Spark SQL
spark.udf.register(
    "format_name_udf",
    format_name_udf
)

customers_df.createOrReplaceTempView(
    "customers"
)

sql_result = spark.sql("""
    SELECT
        customer_id,
        format_name_udf(customer_name)
            AS formatted_name
    FROM customers
""")

# IntegerType example
def name_length(customer_name):
    if customer_name is None:
        return None
    return len(customer_name)

name_length_udf = udf(
    name_length,
    IntegerType()
)
