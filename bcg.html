Employees Earning More Than Their Managers

PostgreSQL
SELECT
    e.employee_id,
    e.employee_name,
    e.salary AS employee_salary,
    m.employee_name AS manager_name,
    m.salary AS manager_salary
FROM employees e
JOIN employees m
    ON e.manager_id = m.employee_id
WHERE e.salary > m.salary;
Pattern: self-join the employees table—once as employee and once as manager.
PySpark
from pyspark.sql import functions as F

e = employees_df.alias("e")
m = employees_df.alias("m")

result_df = (
    e.join(
        m,
        F.col("e.manager_id") == F.col("m.employee_id"),
        "inner"
    )
    .filter(F.col("e.salary") > F.col("m.salary"))
    .select(
        F.col("e.employee_id"),
        F.col("e.employee_name"),
        F.col("e.salary").alias("employee_salary"),
        F.col("m.employee_name").alias("manager_name"),
        F.col("m.salary").alias("manager_salary")
    )
)

Customers Without Orders

PostgreSQL
SELECT
    c.customer_id,
    c.customer_name
FROM customers c
WHERE NOT EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id
);
NOT EXISTS is preferred because it handles right-side NULL values safely.
PySpark
result_df = customers_df.join(
    orders_df.select("customer_id").distinct(),
    on="customer_id",
    how="left_anti"
)
left_anti returns left-side rows that have no match on the right.

Second-Highest Salary

PostgreSQL
WITH ranked AS (
    SELECT
        employee_id,
        employee_name,
        salary,
        DENSE_RANK() OVER (
            ORDER BY salary DESC
        ) AS salary_rank
    FROM employees
    WHERE salary IS NOT NULL
)
SELECT
    employee_id,
    employee_name,
    salary
FROM ranked
WHERE salary_rank = 2;
Use DENSE_RANK() when multiple employees may share the same salary.
PySpark
from pyspark.sql import functions as F
from pyspark.sql.window import Window

salary_window = Window.orderBy(
    F.col("salary").desc()
)

result_df = (
    employees_df
    .filter(F.col("salary").isNotNull())
    .withColumn(
        "salary_rank",
        F.dense_rank().over(salary_window)
    )
    .filter(F.col("salary_rank") == 2)
    .drop("salary_rank")
)

Top N Employees per Department

PostgreSQL
WITH ranked AS (
    SELECT
        employee_id,
        employee_name,
        department,
        salary,
        ROW_NUMBER() OVER (
            PARTITION BY department
            ORDER BY salary DESC, employee_id
        ) AS row_num
    FROM employees
)
SELECT
    employee_id,
    employee_name,
    department,
    salary
FROM ranked
WHERE row_num <= 3;
ROW_NUMBER() returns exactly N rows. Use DENSE_RANK() to include ties.
PySpark
department_window = (
    Window
    .partitionBy("department")
    .orderBy(
        F.col("salary").desc(),
        F.col("employee_id")
    )
)

result_df = (
    employees_df
    .withColumn(
        "row_num",
        F.row_number().over(department_window)
    )
    .filter(F.col("row_num") <= 3)
    .drop("row_num")
)

Remove Duplicate Records — Keep the Latest

PostgreSQL
WITH ranked AS (
    SELECT
        *,
        ROW_NUMBER() OVER (
            PARTITION BY transaction_id
            ORDER BY
                updated_at DESC,
                ingestion_time DESC
        ) AS row_num
    FROM transactions
)
SELECT *
FROM ranked
WHERE row_num = 1;
Use deterministic ordering when multiple versions have the same business key.
PySpark
dedup_window = (
    Window
    .partitionBy("transaction_id")
    .orderBy(
        F.col("updated_at").desc(),
        F.col("ingestion_time").desc()
    )
)

result_df = (
    transactions_df
    .withColumn(
        "row_num",
        F.row_number().over(dedup_window)
    )
    .filter(F.col("row_num") == 1)
    .drop("row_num")
)
Do not use only dropDuplicates(["transaction_id"]) when the latest row must be preserved.

Running Total

PostgreSQL
SELECT
    transaction_id,
    customer_id,
    transaction_date,
    amount,
    SUM(amount) OVER (
        PARTITION BY customer_id
        ORDER BY transaction_date, transaction_id
        ROWS BETWEEN UNBOUNDED PRECEDING
                 AND CURRENT ROW
    ) AS running_total
FROM transactions;
PySpark
running_window = (
    Window
    .partitionBy("customer_id")
    .orderBy(
        F.col("transaction_date"),
        F.col("transaction_id")
    )
    .rowsBetween(
        Window.unboundedPreceding,
        Window.currentRow
    )
)

result_df = transactions_df.withColumn(
    "running_total",
    F.sum("amount").over(running_window)
)

Consecutive Dates — Gaps and Islands

PostgreSQL
WITH distinct_dates AS (
    SELECT DISTINCT
        customer_id,
        activity_date
    FROM customer_activity
),
numbered AS (
    SELECT
        customer_id,
        activity_date,
        ROW_NUMBER() OVER (
            PARTITION BY customer_id
            ORDER BY activity_date
        ) AS row_num
    FROM distinct_dates
),
grouped AS (
    SELECT
        customer_id,
        activity_date,
        activity_date - row_num::int AS date_group
    FROM numbered
)
SELECT
    customer_id,
    MIN(activity_date) AS start_date,
    MAX(activity_date) AS end_date,
    COUNT(*) AS consecutive_days
FROM grouped
GROUP BY customer_id, date_group
ORDER BY customer_id, start_date;
PySpark
date_window = (
    Window
    .partitionBy("customer_id")
    .orderBy("activity_date")
)

numbered_df = (
    activity_df
    .select("customer_id", "activity_date")
    .distinct()
    .withColumn(
        "row_num",
        F.row_number().over(date_window)
    )
    .withColumn(
        "date_group",
        F.expr("date_sub(activity_date, row_num)")
    )
)

result_df = (
    numbered_df
    .groupBy("customer_id", "date_group")
    .agg(
        F.min("activity_date").alias("start_date"),
        F.max("activity_date").alias("end_date"),
        F.count("*").alias("consecutive_days")
    )
    .drop("date_group")
)

Monthly Customer Revenue

PostgreSQL
SELECT
    customer_id,
    DATE_TRUNC('month', order_date)::date
        AS revenue_month,
    SUM(amount) AS monthly_revenue
FROM orders
GROUP BY
    customer_id,
    DATE_TRUNC('month', order_date)::date
ORDER BY
    customer_id,
    revenue_month;
PySpark
result_df = (
    orders_df
    .withColumn(
        "revenue_month",
        F.to_date(
            F.date_trunc("month", F.col("order_date"))
        )
    )
    .groupBy(
        "customer_id",
        "revenue_month"
    )
    .agg(
        F.sum("amount").alias("monthly_revenue")
    )
    .orderBy(
        "customer_id",
        "revenue_month"
    )
)

First and Latest Transaction per Customer

PostgreSQL
WITH ranked AS (
    SELECT
        *,
        ROW_NUMBER() OVER (
            PARTITION BY customer_id
            ORDER BY transaction_date, transaction_id
        ) AS first_rank,
        ROW_NUMBER() OVER (
            PARTITION BY customer_id
            ORDER BY
                transaction_date DESC,
                transaction_id DESC
        ) AS latest_rank
    FROM transactions
)
SELECT *
FROM ranked
WHERE first_rank = 1
   OR latest_rank = 1
ORDER BY customer_id, transaction_date;
PySpark
first_window = (
    Window
    .partitionBy("customer_id")
    .orderBy(
        F.col("transaction_date"),
        F.col("transaction_id")
    )
)

latest_window = (
    Window
    .partitionBy("customer_id")
    .orderBy(
        F.col("transaction_date").desc(),
        F.col("transaction_id").desc()
    )
)

result_df = (
    transactions_df
    .withColumn(
        "first_rank",
        F.row_number().over(first_window)
    )
    .withColumn(
        "latest_rank",
        F.row_number().over(latest_window)
    )
    .filter(
        (F.col("first_rank") == 1)
        | (F.col("latest_rank") == 1)
    )
)

Find Duplicate Business Keys

PostgreSQL
SELECT
    order_id,
    COUNT(*) AS duplicate_count
FROM orders
GROUP BY order_id
HAVING COUNT(*) > 1;
For a composite key, include every key column in both SELECT and GROUP BY.
PySpark
duplicate_keys_df = (
    orders_df
    .groupBy("order_id")
    .agg(
        F.count("*").alias("duplicate_count")
    )
    .filter(F.col("duplicate_count") > 1)
)

duplicate_rows_df = orders_df.join(
    duplicate_keys_df.select("order_id"),
    on="order_id",
    how="inner"
)

NOT EXISTS vs NOT IN vs Left Anti Join

PostgreSQL
-- Preferred
SELECT c.*
FROM customers c
WHERE NOT EXISTS (
    SELECT 1
    FROM orders o
    WHERE o.customer_id = c.customer_id
);

-- Use carefully: NULL can cause unexpected results
SELECT c.*
FROM customers c
WHERE c.customer_id NOT IN (
    SELECT o.customer_id
    FROM orders o
    WHERE o.customer_id IS NOT NULL
);
MethodNULL-safe?Recommendation
NOT EXISTSYesPreferred
NOT INNo, unless NULLs removedUse carefully
LEFT JOIN ... IS NULLYesValid alternative
PySpark
result_df = customers_df.join(
    orders_df.select("customer_id").distinct(),
    on="customer_id",
    how="left_anti"
)
In PySpark, left_anti is the direct equivalent of SQL anti-join logic.

Find Customers Whose January Order Total Is Greater Than 10,000

PostgreSQL
SELECT
    c.customer_id,
    c.customer_name,
    SUM(o.amount) AS total_amount
FROM customers c
INNER JOIN orders o
    ON c.customer_id = o.customer_id
WHERE o.order_date >= DATE '2026-01-01'
  AND o.order_date <  DATE '2026-02-01'
GROUP BY
    c.customer_id,
    c.customer_name
HAVING SUM(o.amount) > 10000
ORDER BY total_amount DESC;
Use WHERE to filter individual January orders and HAVING to filter the aggregated customer total.
PySpark
from pyspark.sql import functions as F

result_df = (
    customers_df.alias("c")
    .join(
        orders_df.alias("o"),
        F.col("c.customer_id") == F.col("o.customer_id"),
        "inner"
    )
    .filter(
        (F.col("o.order_date") >=
         F.lit("2026-01-01").cast("date"))
        &
        (F.col("o.order_date") <
         F.lit("2026-02-01").cast("date"))
    )
    .groupBy(
        F.col("c.customer_id"),
        F.col("c.customer_name")
    )
    .agg(
        F.sum("o.amount").alias("total_amount")
    )
    .filter(F.col("total_amount") > 10000)
    .orderBy(F.col("total_amount").desc())
)

Character Frequency and First Occurrence Index

Python
def character_details(text: str) -> dict[str, dict[str, int]]:
    result = {}

    for index, char in enumerate(text):
        if char not in result:
            result[char] = {
                "frequency": 1,
                "first_index": index
            }
        else:
            result[char]["frequency"] += 1

    return result


print(character_details("banana"))
Time complexity: O(n). Space complexity: O(k), where k is the number of unique characters.

Check Whether a String Is a Palindrome

Python
def is_palindrome(value: str) -> bool:
    cleaned = "".join(
        char.lower()
        for char in value
        if char.isalnum()
    )

    left = 0
    right = len(cleaned) - 1

    while left < right:
        if cleaned[left] != cleaned[right]:
            return False

        left += 1
        right -= 1

    return True


print(is_palindrome("A man, a plan, a canal: Panama"))
# True
The two-pointer comparison is O(n). Creating cleaned also uses O(n) extra space.

Abstract Product and PerishableProduct with Expiry Discount

Python
from abc import ABC, abstractmethod
from datetime import date


class Product(ABC):
    def __init__(self, name: str, price: float) -> None:
        if price < 0:
            raise ValueError("Price cannot be negative")

        self.name = name
        self.price = price

    @abstractmethod
    def calculate_price(self, today: date | None = None) -> float:
        pass


class PerishableProduct(Product):
    DISCOUNT_RATE = 0.20
    EXPIRY_THRESHOLD_DAYS = 3

    def __init__(
        self,
        name: str,
        price: float,
        expiry_date: date
    ) -> None:
        super().__init__(name, price)
        self.expiry_date = expiry_date

    def calculate_price(self, today: date | None = None) -> float:
        today = today or date.today()
        days_remaining = (self.expiry_date - today).days

        if days_remaining < 0:
            return 0.0

        if days_remaining <= self.EXPIRY_THRESHOLD_DAYS:
            return self.price * (1 - self.DISCOUNT_RATE)

        return self.price


milk = PerishableProduct(
    name="Milk",
    price=100.0,
    expiry_date=date(2026, 7, 16)
)

print(milk.calculate_price(date(2026, 7, 14)))
# 80.0
Product cannot be instantiated directly because it has an abstract method. super() initializes the shared product fields.
