WITH process_times AS (
    SELECT
        machine_id,
        process_id,
        MAX(timestamp) FILTER (
            WHERE activity_type = 'end'
        ) -
        MAX(timestamp) FILTER (
            WHERE activity_type = 'start'
        ) AS processing_time
    FROM Activity
    GROUP BY machine_id, process_id
)
SELECT
    machine_id,
    ROUND(
        AVG(processing_time)::numeric,
        3
    ) AS processing_time
FROM process_times
GROUP BY machine_id;
PySpark DataFrame API 
process_times = (
    activity_df
    .groupBy("machine_id", "process_id")
    .agg(
        F.max(
            F.when(
                F.col("activity_type") == "start",
                F.col("timestamp")
            )
        ).alias("start_time"),
        F.max(
            F.when(
                F.col("activity_type") == "end",
                F.col("timestamp")
            )
        ).alias("end_time"),
    )
    .withColumn(
        "processing_time",
        F.col("end_time") - F.col("start_time")
    )
)

result = (
    process_times
    .groupBy("machine_id")
    .agg(
        F.round(
            F.avg("processing_time"),
            3
        ).alias("processing_time")
    )
)
pandas 
pivoted = activity.pivot_table(
    index=["machine_id", "process_id"],
    columns="activity_type",
    values="timestamp",
    aggfunc="max",
).reset_index()

pivoted["processing_time"] = (
    pivoted["end"] - pivoted["start"]
)

result = (
    pivoted.groupby("machine_id", as_index=False)
    .agg(processing_time=("processing_time", "mean"))
)

result["processing_time"] = (
    result["processing_time"].round(3)

        AS attended_exams

    ON e.student_id = s.student_id
   AND e.subject_name = sub.subject_name

    students_df
    .crossJoin(subjects_df)
    .join(
        exam_counts,

    .fillna({"attended_exams": 0})
    .select(
        "student_id",
        "student_name",
        "subject_name",

16. 1251 — Average Selling Price

LeetCode ↗ ↑ Index
SQL · PostgreSQL 
SELECT
    p.product_id,
    ROUND(
        COALESCE(
            SUM(p.price * u.units)::numeric /
            NULLIF(SUM(u.units), 0),
            0
        ),
        2
    ) AS average_price
FROM Prices p
LEFT JOIN UnitsSold u
    ON u.product_id = p.product_id
   AND u.purchase_date
       BETWEEN p.start_date AND p.end_date
GROUP BY p.product_id;
PySpark DataFrame API 
p = prices_df.alias("p")
u = units_sold_df.alias("u")

joined = p.join(
    u,
    (
        F.col("p.product_id") ==
        F.col("u.product_id")
    ) &
    F.col("u.purchase_date").between(
        F.col("p.start_date"),
        F.col("p.end_date")
    ),
    "left"
)

result = (
    joined
    .groupBy(F.col("p.product_id"))
    .agg(
        F.round(
            F.coalesce(
                F.sum(
                    F.col("p.price") *
                    F.col("u.units")
                ) / F.sum("u.units"),
                F.lit(0.0)
            ),
            2
        ).alias("average_price")
    )
    .select(
        F.col("p.product_id").alias("product_id"),
        "average_price"
    )
)
pandas 
matched = prices.merge(
    units_sold,
    on="product_id",
    how="inner",
)

matched = matched.loc[
    matched["purchase_date"].between(
        matched["start_date"],
        matched["end_date"],
    )
].copy()

matched["revenue"] = (
    matched["price"] * matched["units"]
)

aggregated = (
    matched.groupby("product_id", as_index=False)
    .agg(
        revenue=("revenue", "sum"),
        units=("units", "sum"),
    )
)

aggregated["average_price"] = (
    aggregated["revenue"] / aggregated["units"]
).round(2)

result = (
    prices[["product_id"]]
    .drop_duplicates()
    .merge(
        aggregated[["product_id", "average_price"]],
        on="product_id",
        how="left",
    )
    .fillna({"average_price": 0})
)

    country;

        "month",
        F.date_format("trans_date", "yyyy-MM")

        F.sum("amount").alias("trans_total_amount"),
        F.sum(
            F.when(
                F.col("state") == "approved",
                F.col("amount")
            ).otherwise(0)
        ).alias("approved_total_amount")
    )
)

    work["trans_date"]

    work["trans_date"].dt.strftime("%Y-%m")

    work["state"].eq("approved"),

work["is_approved"] = (
    work["state"].eq("approved").astype(int)

    work.groupby(
        ["month", "country"],

        as_index=False,
    )
