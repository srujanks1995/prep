from collections import Counter

def minimum_window(text, pattern):
    if not text or not pattern:
        return ""

    required = Counter(pattern)
    window = {}
    formed = 0
    needed = len(required)
    left = 0
    best_start = 0
    best_length = float("inf")

    for right, char in enumerate(text):
        window[char] = window.get(char, 0) + 1

        if char in required and window[char] == required[char]:
            formed += 1

        while formed == needed:
            if right - left + 1 < best_length:
                best_start = left
                best_length = right - left + 1

            left_char = text[left]
            window[left_char] -= 1
            if left_char in required and window[left_char] < required[left_char]:
                formed -= 1
            left += 1

    if best_length == float("inf"):
        return ""
    return text[best_start:best_start + best_length]

def quicksort(values):
    if len(values) <= 1:
        return values.copy()

    pivot = values[len(values) // 2]
    smaller = [value for value in values if value < pivot]
    equal = [value for value in values if value == pivot]
    greater = [value for value in values if value > pivot]

    return quicksort(smaller) + equal + quicksort(greater)

def spiral_order(matrix):
    if not matrix:
        return []

    result = []
    top, bottom = 0, len(matrix) - 1
    left, right = 0, len(matrix[0]) - 1

    while top <= bottom and left <= right:
        for column in range(left, right + 1):
            result.append(matrix[top][column])
        top += 1

        for row in range(top, bottom + 1):
            result.append(matrix[row][right])
        right -= 1

        if top <= bottom:

                result.append(matrix[bottom][column])
            bottom -= 1

        if left <= right:
            for row in range(bottom, top - 1, -1):

def fill_missing_with_group_average(
    records,
    group_field,

):

    counts = {}

        group = record[group_field]
