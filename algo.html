class Solution:
    def minWindow(self, s, t):
        if not t:
            return ''
        need = Counter(t)

        left = 0

        write = left

            if nums[i] <= pivot:
                nums[write], nums[i] = nums[i], nums[write]

        return write

        while left < right:
            pivot = partition(left, right)
            if pivot - left < right - pivot:
                sort(left, pivot - 1)
                left = pivot + 1
            else:
                sort(pivot + 1, right)
                right = pivot - 1

    return nums

            largest = root

            right = left + 1
